// Função para calcular a área de um retângulo
FUNCAO areaRetangulo(lado1, lado2)
    RETORNA lado1 * lado2
FIM FUNCAO

// Função para calcular o perímetro de um retângulo
FUNCAO perimetroRetangulo(lado1, lado2)
    RETORNA 2 * (lado1 + lado2)
FIM FUNCAO

// Função para calcular a hipotenusa
FUNCAO hipotenusaRetangulo(lado1, lado2)
    // A função POW é para calcular potência, e SQRT para raiz quadrada.
    RETORNA SQRT( (lado1 * lado1) + (lado2 * lado2) )
FIM FUNCAO

// Início do programa principal
PRINCIPAL
    // Declaração de variáveis
    INTEIRO comprimento, largura

    // Entrada de dados
    ESCREVER "Digite o comprimento do retângulo: "
    LER comprimento

    ESCREVER "Digite a largura do retângulo: "
    LER largura

    // Chamada das funções e exibição dos resultados
    ESCREVER "Área = ", areaRetangulo(comprimento, largura)
    ESCREVER "Perímetro = ", perimetroRetangulo(comprimento, largura)
    ESCREVER "Hipotenusa = ", hipotenusaRetangulo(comprimento, largura)

FIM PRINCIPAL
