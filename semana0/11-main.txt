// Declaração de variáveis
REAL principal, taxa, tempo, montante, jurosCompostos
CARACTERE tipoTaxa, tipoTempo

INICIO

    // Entrada de dados do usuário
    ESCREVER "Digite o valor principal: "
    LER principal

    ESCREVER "A taxa é anual (a) ou mensal (m)? "
    LER tipoTaxa

    ESCREVER "Digite a taxa de juros (%): "
    LER taxa

    ESCREVER "O tempo é em anos (a) ou meses (m)? "
    LER tipoTempo

    ESCREVER "Digite o tempo: "
    LER tempo

    // Ajusta taxa e tempo para o cálculo correto, se necessário
    SE (tipoTaxa == 'a' E tipoTempo == 'm') ENTÃO
        taxa = taxa / 12
        tempo = tempo / 12
    SENÃO SE (tipoTaxa == 'm' E tipoTempo == 'a') ENTÃO
        tempo = tempo * 12
    FIM SE

    // Cálculo do montante
    montante = principal * ( (1 + taxa / 100) ^ tempo )

    // Cálculo dos juros compostos
    jurosCompostos = montante - principal

    // Exibição do resultado
    ESCREVER "Juros Compostos = ", jurosCompostos

FIM
